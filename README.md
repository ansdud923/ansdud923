#### 🐧 Linux 명령어 학습
```bash
# 매일 연습하는 기본 명령어들
ps aux | grep java          # 프로세스 모니터링
df -h                       # 디스크 사용량 확인
top -p $(pgrep java)        # CPU/메모리 사용률
tail -f /var/log/syslog     # 실시간 로그 모니터링
systemctl status nginx     # 서비스 상태 확인
```
